# 1. 로컬 저장소 초기화 (폴더 내부에 .git 폴더 생성)
git init

# 2. 원격 저장소 주소 등록
git remote add origin 원격 저장소 주소

# 3. 현재 연결된 저장소 확인
git remote -v

# 현재 상태 확인 (수정된 파일 등)
git status

# 1. 스테이징 영역에 파일 추가 (전체 추가: . / 특정 파일: 파일명)
git add .

# 브랜치 이름 메인으로 변경
git branch -M main

# 2. 로컬 저장소에 저장 (커밋 메시지는 필수)
git commit -m "커밋 메시지 입력"

# 3. GitHub 원격 저장소로 업로드
# 최초 1회: git push -u origin main (또는 master)
git push origin main
git push -u origin main

# 4. GitHub의 최신 코드를 로컬로 가져오기
git pull origin main
